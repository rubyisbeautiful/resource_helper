      def plural_url(params=nil, escape=true)
        tmp = site.dup
        unless params.blank?
          query = ''
          params.each_pair do |k,v|
            if escape
              query << CGI.escape("#{k}=#{v}&")
            else
              query << "#{k}=#{v}"
            end
          end
        end
        tmp.path = collection_path
        tmp.query = query
        return tmp
      end
